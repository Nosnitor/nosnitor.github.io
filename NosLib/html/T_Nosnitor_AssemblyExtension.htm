AssemblyExtension Class

Extends the System.Reflection.Assembly class.

The methods in this class return information available in the assembly manifest.

Examples
Showing values of the calling assembly:
C#
VB
Copy
// Assembly Identity Attributes
string version = Nosnitor.AssemblyExtension.GetAssemblyVersion();

// Assembly Informational Attributes
string company = Nosnitor.AssemblyExtension.GetAssemblyCompany();
string copyright = Nosnitor.AssemblyExtension.GetAssemblyCopyright();
string fileVersion = Nosnitor.AssemblyExtension.GetAssemblyFileVersion();
string informationalVersion = Nosnitor.AssemblyExtension.GetAssemblyInformationalVersion();
string product = Nosnitor.AssemblyExtension.GetAssemblyInformationalVersion();
string trademark = Nosnitor.AssemblyExtension.GetAssemblyTrademark();

// Assembly Manifest Attributes
string configuration = Nosnitor.AssemblyExtension.GetAssemblyConfiguration();
string defaultAlias = Nosnitor.AssemblyExtension.GetAssemblyDefaultAlias();
string description = Nosnitor.AssemblyExtension.GetAssemblyDescription();
string title = Nosnitor.AssemblyExtension.GetAssemblyTitle();

string assemblyPath = System.IO.Directory.GetCurrentDirectory() + "\\" + "TestAssembly.dll";

Dim assembly as System.Reflection.Assembly = System.Reflection.Assembly.LoadFile(assemblyPath)
