byte[] array1 = new byte[] { 20, 21, 22, 23 };
byte[] array2 = new byte[] { 20, 21, 22, 23 };
byte[] array3 = new byte[] { 20, 21, 22, 24 };
byte[] array4 = new byte[] { 20, 21, 22 };

bool comp;
comp = Nosnitor.Comparison.ByteArrayCompare(array1, array2);
// Returns True
comp = Nosnitor.Comparison.ByteArrayCompare(array1, array3);
// Returns False
comp = Nosnitor.Comparison.ByteArrayCompare(array1, array4);
// Returns False
comp = Nosnitor.Comparison.ByteArrayCompare(array1, null);
// Returns False
comp = Nosnitor.Comparison.ByteArrayCompare(null, null);
// Returns True
